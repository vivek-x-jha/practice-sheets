\section*{Instructions}
Solve each problem. Write exact answers; simplify fractions. For percent problems, show the final answer as a number with units when appropriate. For variable expressions, simplify fully. Not timed.

\section*{A. Percentages}
\begin{enumerate}
  \item Find 15\% of 240.
  \item 72 is what percent of 96?
  \item Increase 80 by 12.5\%.
  \item Decrease 150 by 30\%.
  \item A shirt costs \$45 and is discounted by 20\%. What is the sale price?
  \item A bill is \$64. Add 8.5\% tax. What is the total?
\end{enumerate}

\section*{B. Ratios and Rates}
\begin{enumerate}
  \setcounter{enumi}{6}
  \item Simplify the ratio $18:24$.
  \item A mix uses a ratio of $3:5$ (red to blue). If there are 40 total parts, how many parts are blue?
  \item A recipe uses 2 cups of sugar for 5 cups of flour. How many cups of sugar are needed for 20 cups of flour?
  \item A car travels 210 miles in 3.5 hours. Find the unit rate in miles per hour.
  \item The ratio of boys to girls is $7:9$. If there are 144 students total, how many girls are there?
  \item A store sells 8 items for \$14.40. What is the unit price per item?
\end{enumerate}

\section*{C. Exponents}
\begin{enumerate}
  \setcounter{enumi}{12}
  \item Simplify $2^3 \cdot 2^5$.
  \item Simplify $\dfrac{3^7}{3^4}$.
  \item Simplify $(4^2)^3$.
  \item Simplify $5^0 + 2^3$.
  \item Simplify $\dfrac{2^6 \cdot 5^2}{10^2}$.
\end{enumerate}

\section*{D. Variable Expressions}
\begin{enumerate}
  \setcounter{enumi}{17}
  \item Simplify $3x + 7x - 5$.
  \item Simplify $4a - 3(2a - 5)$.
  \item Simplify $\dfrac{2x^2y}{4xy^2}$.
\end{enumerate}
