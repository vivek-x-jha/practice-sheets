\section*{Instructions}
Solve each problem. For fraction problems, \textbf{write your final answer as an improper fraction}. For pre-algebra items, simplify fully.

\section*{A. Addition and Subtraction of Fractions}
\begin{enumerate}
  \item $\frac{7}{12} + \frac{5}{8}$
  \item $\frac{13}{15} - \frac{7}{10}$
  \item $\frac{9}{14} + \frac{3}{7}$
  \item $\frac{5}{6} - \frac{11}{12}$
  \item $\frac{2}{9} + \frac{7}{18}$
  \item $\frac{17}{20} - \frac{3}{5}$
\end{enumerate}

\section*{B. Multiplying and Dividing Fractions}
\begin{enumerate}
  \setcounter{enumi}{6}
  \item $\frac{4}{9} \times \frac{15}{16}$
  \item $\frac{7}{8} \times \frac{6}{5}$
  \item $\frac{5}{12} \div \frac{3}{8}$
  \item $\frac{11}{15} \div \frac{22}{45}$
  \item $3\frac{1}{4} \times \frac{2}{3}$
  \item $5 \div \frac{7}{10}$
\end{enumerate}

\section*{C. Mixed Numbers and Combined Operations}
\begin{enumerate}
  \setcounter{enumi}{12}
  \item $1\frac{5}{6} + \frac{2}{3}$
  \item $3\frac{1}{4} - 1\frac{5}{8}$
  \item $\left(\frac{5}{6} + \frac{1}{4}\right) - \frac{2}{3}$
  \item $\left(2\frac{1}{5} \div \frac{3}{10}\right) + \frac{1}{2}$
\end{enumerate}

\section*{D. Pre-Algebra Basics}
\begin{enumerate}
  \setcounter{enumi}{16}
  \item Simplify: $3(4-7) + 2^3$
  \item Solve: $5x - 7 = 18$
  \item Solve: $2(x-3) = 4x + 6$
  \item Evaluate: $-3^2 + (-4)(2) + 15$
\end{enumerate}
