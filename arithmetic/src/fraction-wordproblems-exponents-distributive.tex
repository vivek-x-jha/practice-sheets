\section*{Instructions}
Solve each problem. For fraction word problems, \textbf{write your final answer as an improper fraction}. For exponent and distributive law problems, simplify fully.

\section*{A. Multi-Step Fraction Word Problems}
\begin{enumerate}
  \item A recipe uses $\tfrac{3}{4}$ cup of sugar per batch. You make $2\frac{1}{2}$ batches and then spill $\tfrac{1}{3}$ cup. How much sugar remains?
  \item A runner completes $\tfrac{5}{6}$ mile in the morning and $\tfrac{3}{4}$ mile in the afternoon. How many miles total? Then subtract $\tfrac{1}{8}$ mile for a cool-down.
  \item A tank is $\tfrac{2}{3}$ full. After using $\tfrac{1}{4}$ of the tank, you add $\tfrac{3}{8}$ of a tank. What fraction of the tank is full now?
  \item A rope is $6\frac{1}{2}$ meters long. You cut off $\tfrac{5}{6}$ meter and then another $\tfrac{3}{4}$ meter. How much rope is left?
  \item A baker uses $\tfrac{2}{5}$ of a bag of flour on Monday and $\tfrac{3}{10}$ on Tuesday. What fraction of the bag is left?
  \item A class walked $\tfrac{7}{8}$ mile on Monday and $\tfrac{2}{3}$ mile on Tuesday. If the total goal is $2$ miles, how much farther do they need to walk?
  \item A jar has $1\frac{1}{3}$ liters of juice. You pour out $\tfrac{2}{5}$ liter and then add $\tfrac{3}{4}$ liter. How many liters are in the jar?
  \item A cyclist rides $\tfrac{5}{12}$ hour at one speed and $\tfrac{7}{18}$ hour at another. What is the total time? Then divide by $\tfrac{2}{3}$ to find how many $\tfrac{2}{3}$-hour blocks were ridden.
  \item A container holds $\tfrac{3}{5}$ pound of nuts. You add $\tfrac{2}{3}$ pound, then split the total equally into 4 bags. How much is in each bag?
  \item A water jug starts with $2\frac{1}{4}$ gallons. You use $\tfrac{3}{8}$ gallon, then refill $\tfrac{5}{6}$ gallon. How much water is in the jug?
\end{enumerate}

\section*{B. Exponent Rules and Distributive Law}
\begin{enumerate}
  \setcounter{enumi}{10}
  \item Simplify $2^4 \cdot 2^3$.
  \item Simplify $\dfrac{5^7}{5^3}$.
  \item Simplify $(3^2)^3$.
  \item Simplify $4^0 + 2^5$.
  \item Simplify $\dfrac{2^6 \cdot 3^2}{6^2}$.
  \item Use the distributive law: $7(12-5)$.
  \item Use the distributive law: $6(4+9)$.
  \item Simplify $3(x+5) - 2(x-4)$.
  \item Simplify $4(2x-3) + 5(x+1)$.
  \item Simplify $2(a-3) - 3(2a+1)$.
\end{enumerate}
