\cheatsheettitle{Algebra I Quadratics Cheatsheet}{Module 9}

\section*{Forms}
Standard form:
\[
y=ax^2+bx+c
\]
Factored form:
\[
y=a(x-r_1)(x-r_2)
\]
Vertex form:
\[
y=a(x-h)^2+k
\]

\section*{Graph Features}
Vertex:
\[
(h,k)\text{ from vertex form}
\]
Axis of symmetry:
\[
x=h
\]
Opening:
\[
a>0\Rightarrow \text{opens up},\qquad a<0\Rightarrow \text{opens down}
\]
\[
|a|>1\Rightarrow \text{narrower},\qquad 0<|a|<1\Rightarrow \text{wider}
\]

\section*{Factoring to Solve}
\[
ax^2+bx+c=0
\]
Factor and use zero product property:
\[
AB=0\Rightarrow A=0\text{ or }B=0
\]

\section*{Pure Quadratics}
\[
ax^2+c=0
\]
Isolate $x^2$, then square root both sides:
\[
x^2=25\Rightarrow x=\pm5
\]

\section*{Quadratic Formula}
\[
ax^2+bx+c=0
\]
\[
x=\frac{-b\pm\sqrt{b^2-4ac}}{2a}
\]

\section*{Discriminant}
\[
D=b^2-4ac
\]
\[
\begin{array}{ll}
D>0 & \text{two real solutions}\\
D=0 & \text{one real solution}\\
D<0 & \text{no real solutions}
\end{array}
\]

\section*{Intercepts}
\[
y\text{-intercept: }(0,c)
\]
\[
x\text{-intercepts: solve }ax^2+bx+c=0
\]

\section*{Quick Checks}
\begin{itemize}
  \item Set the quadratic equal to zero before factoring to solve.
  \item Include both signs when taking square roots.
  \item Vertex gives maximum if $a<0$ and minimum if $a>0$.
  \item Use exact radical form unless decimals are requested.
\end{itemize}
