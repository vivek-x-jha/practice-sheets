\cheatsheettitle{Algebra I Polynomials Intro Cheatsheet}{Module 8}

\section*{Vocabulary}
\[
\text{monomial: one term},\qquad
\text{binomial: two terms},\qquad
\text{trinomial: three terms}
\]
Degree of a term:
\[
3x^4 \Rightarrow \text{degree }4
\]
Degree of a polynomial is the greatest term degree.

\section*{Adding and Subtracting}
Combine like terms.
\[
(3x^2+5x-1)+(2x^2-x+7)=5x^2+4x+6
\]
Distribute the negative when subtracting:
\[
(4x^2-3x)-(x^2+6x-2)=3x^2-9x+2
\]

\section*{Multiplying}
Monomial times polynomial:
\[
a(b+c+d)=ab+ac+ad
\]
Binomial product:
\[
(a+b)(c+d)=ac+ad+bc+bd
\]

\section*{Special Products}
\[
(a+b)^2=a^2+2ab+b^2
\]
\[
(a-b)^2=a^2-2ab+b^2
\]
\[
(a+b)(a-b)=a^2-b^2
\]

\section*{Factoring GCF}
Find the greatest common factor of all terms.
\[
6x^3+9x^2=3x^2(2x+3)
\]

\section*{Factoring Trinomials}
For
\[
x^2+bx+c
\]
find two numbers that multiply to $c$ and add to $b$.
\[
x^2+7x+12=(x+3)(x+4)
\]

\section*{Factoring by Grouping}
\[
ax+ay+bx+by
\]
\[
=a(x+y)+b(x+y)
\]
\[
=(a+b)(x+y)
\]

\section*{Quick Checks}
\begin{itemize}
  \item Put polynomials in descending degree order.
  \item Factor out the GCF first.
  \item Check factoring by multiplying back.
  \item Difference of squares needs subtraction between two square terms.
\end{itemize}
