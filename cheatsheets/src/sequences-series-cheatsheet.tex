\section*{Core Definitions}
\[
\{a_n\}=\text{sequence},\qquad
S_n=a_1+a_2+\cdots+a_n
\]

Recursive vs explicit:
\[
\text{recursive: } a_n \text{ from previous terms},\qquad
\text{explicit: } a_n \text{ directly in terms of } n
\]

\section*{Arithmetic Sequences and Series}
\[
a_n=a_1+(n-1)d
\]
\[
S_n=\frac{n}{2}(a_1+a_n)=\frac{n}{2}\bigl(2a_1+(n-1)d\bigr)
\]

\section*{Geometric Sequences and Series}
\[
a_n=a_1r^{\,n-1}
\]
\[
S_n=\frac{a_1(1-r^n)}{1-r}\quad (r\ne 1)
\]
\[
S_\infty=\frac{a_1}{1-r}\quad \text{for } |r|<1
\]

\section*{Sigma Facts}
\[
\sum_{k=1}^n c=cn,\qquad
\sum_{k=1}^n k=\frac{n(n+1)}{2}
\]
\[
\sum_{k=1}^n k^2=\frac{n(n+1)(2n+1)}{6},\qquad
\sum_{k=1}^n k^3=\left(\frac{n(n+1)}{2}\right)^2
\]

\section*{Convergence Reminders}
\begin{itemize}
  \item If $a_n$ approaches a nonzero number or does not approach a limit, then $\sum a_n$ diverges.
  \item A geometric series converges only when $|r|<1$.
  \item Telescoping series often collapse after partial fraction decomposition.
\end{itemize}
