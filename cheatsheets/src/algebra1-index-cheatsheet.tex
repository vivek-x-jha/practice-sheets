\cheatsheettitle{Algebra I Module Index}{Prerequisite Map}

\section*{How to Use This Index}
Use the modules in order when rebuilding foundations. Use individual modules as quick reference sheets before practice, quizzes, or corrections.

\section*{Module Sequence}
\begin{center}
\begin{tabular}{clp{3.6in}}
\toprule
Order & Module & Main Skills \\
\midrule
1 & Foundations & Arithmetic structure, signed numbers, fractions, exponents, radicals, evaluating and simplifying expressions \\
2 & Linear Equations & Multi-step equations, distributive property, variables on both sides, literal equations, absolute value equations \\
3 & Inequalities & One-variable inequalities, compound inequalities, interval notation, absolute value inequalities \\
4 & Proportional Relationships & Ratios, unit rates, percent change, direct variation, slope as rate of change \\
5 & Linear Functions & Slope, intercepts, graphing lines, slope-intercept, point-slope, standard form \\
6 & Systems & Graphing, substitution, elimination, interpreting intersections \\
7 & Exponents and Radicals & Exponent rules, scientific notation, square roots, simplifying radicals, rational and irrational numbers \\
8 & Polynomials Intro & Polynomial operations, special products, GCF, trinomial factoring \\
9 & Quadratics & Factoring, pure quadratics, quadratic formula, vertex form, graph features, applications \\
10 & Functions and Modeling & Function notation, domain and range, tables, arithmetic sequences, simple exponential growth \\
11 & Statistics and Probability & Center, spread, box plots, two-way tables, probability, compound probability \\
\bottomrule
\end{tabular}
\end{center}

\section*{Prerequisite Flow}
\[
\text{Foundations}\rightarrow \text{Equations}\rightarrow \text{Inequalities}\rightarrow \text{Proportional Relationships}\rightarrow \text{Linear Functions}
\]
\[
\text{Linear Functions}\rightarrow \text{Systems}\rightarrow \text{Polynomials}\rightarrow \text{Quadratics}
\]
\[
\text{Functions and Modeling}\rightarrow \text{Statistics and Probability}
\]

\section*{Readiness Checks}
\begin{itemize}
  \item Before linear functions: know equation solving, slope, and graph coordinates.
  \item Before systems: know how to graph and write linear equations.
  \item Before quadratics: know factoring, square roots, and function features.
  \item Before statistics and probability: know fractions, decimals, percents, and table interpretation.
\end{itemize}
