\section*{Equation Solving}
\[
a(x+b)+c=dx+e
\]
Distribute, combine like terms, collect variables, collect constants, divide by the coefficient.

\section*{Inequalities}
\[
-ax+b<c
\]
Flip the inequality sign when multiplying or dividing by a negative number.
\[
|A|<c\Rightarrow -c<A<c
\]
\[
|A|>c\Rightarrow A<-c\text{ or }A>c
\]

\section*{Factoring Toolkit}
\[
a^2-b^2=(a-b)(a+b)
\]
\[
a^2\pm2ab+b^2=(a\pm b)^2
\]
\[
a^3-b^3=(a-b)(a^2+ab+b^2)
\]
\[
a^3+b^3=(a+b)(a^2-ab+b^2)
\]
\begin{itemize}
  \item Factor out the GCF first.
  \item Use grouping for four terms.
  \item For $ax^2+bx+c$, look for factors whose product is $ac$ and sum is $b$.
  \item For $ax^4+bx^2+c$, substitute $u=x^2$.
\end{itemize}

\section*{Radicals and Rational Exponents}
\[
a^{1/n}=\sqrt[n]{a}
\]
\[
a^{m/n}=\sqrt[n]{a^m}=(\sqrt[n]{a})^m
\]
\[
\sqrt{ab}=\sqrt a\sqrt b,\qquad
\sqrt{\frac ab}=\frac{\sqrt a}{\sqrt b}
\]

\section*{Function Notation}
\[
f(a)\text{ means substitute }x=a
\]
\[
(f+g)(x)=f(x)+g(x)
\]
\[
(fg)(x)=f(x)g(x)
\]
\[
\left(\frac fg\right)(x)=\frac{f(x)}{g(x)}\qquad g(x)\ne0
\]

\section*{Restrictions to Check}
\begin{itemize}
  \item Denominators cannot equal $0$.
  \item Even roots require nonnegative radicands over the reals.
  \item Logarithm arguments must be positive.
  \item Squaring both sides can create extraneous solutions.
\end{itemize}
