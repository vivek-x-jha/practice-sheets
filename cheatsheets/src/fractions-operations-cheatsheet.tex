\section*{Basic Forms}
\[
a\frac{b}{c}=\frac{ac+b}{c}
\]
\[
\frac{a}{b},\qquad b\ne 0
\]

\section*{Equivalent Fractions}
\[
\frac{a}{b}=\frac{ak}{bk}\qquad (k\ne 0)
\]

\section*{Adding and Subtracting}
Find a common denominator first.
\[
\frac{a}{b}+\frac{c}{d}=\frac{ad+bc}{bd}
\]
\[
\frac{a}{b}-\frac{c}{d}=\frac{ad-bc}{bd}
\]

\section*{Multiplying}
\[
\frac{a}{b}\cdot \frac{c}{d}=\frac{ac}{bd}
\]
Cross-cancel before multiplying when possible.

\section*{Dividing}
\[
\frac{a}{b}\div \frac{c}{d}=\frac{a}{b}\cdot \frac{d}{c}=\frac{ad}{bc}
\]

\section*{LCM and LCD}
\begin{itemize}
  \item LCD = least common denominator
  \item find the LCM of the denominators
  \item rewrite each fraction with that denominator
\end{itemize}

\section*{Signs}
\[
\frac{-a}{b}=-\frac{a}{b}=\frac{a}{-b}
\]
\[
\frac{-a}{-b}=\frac{a}{b}
\]

\section*{Order of Operations}
\begin{itemize}
  \item simplify inside grouping symbols first
  \item multiplication and division before addition and subtraction
  \item treat a mixed number carefully in multiplication
\end{itemize}
