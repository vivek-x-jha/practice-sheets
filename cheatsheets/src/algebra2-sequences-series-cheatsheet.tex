\cheatsheettitle{Algebra II Sequences and Series Cheatsheet}{Module 8}

\section*{Sequence vs Series}
\[
\text{sequence: list of terms}
\]
\[
\text{series: sum of terms}
\]

\section*{Arithmetic Sequences}
\[
a_n=a_1+(n-1)d
\]
\[
d=a_n-a_{n-1}
\]
Recursive form:
\[
a_n=a_{n-1}+d,\qquad a_1=\text{first term}
\]

\section*{Arithmetic Series}
\[
S_n=\frac{n(a_1+a_n)}2
\]
\[
S_n=\frac n2\left(2a_1+(n-1)d\right)
\]

\section*{Geometric Sequences}
\[
a_n=a_1r^{n-1}
\]
\[
r=\frac{a_n}{a_{n-1}}
\]
Recursive form:
\[
a_n=ra_{n-1},\qquad a_1=\text{first term}
\]

\section*{Geometric Series}
\[
S_n=\frac{a_1(1-r^n)}{1-r}\qquad (r\ne1)
\]
Infinite geometric series:
\[
S_\infty=\frac{a_1}{1-r}\qquad (|r|<1)
\]
If $|r|\ge1$, the infinite geometric series diverges.

\section*{Sigma Notation}
\[
\sum_{k=1}^{n} a_k=a_1+a_2+\cdots+a_n
\]
\[
\sum_{k=1}^{n} c=cn
\]
\[
\sum_{k=1}^{n} (a_k+b_k)=\sum_{k=1}^{n}a_k+\sum_{k=1}^{n}b_k
\]

\section*{Quick Checks}
\begin{itemize}
  \item Arithmetic means constant difference.
  \item Geometric means constant ratio.
  \item Infinite geometric sums require $|r|<1$.
  \item Match the starting index to the first term.
\end{itemize}
