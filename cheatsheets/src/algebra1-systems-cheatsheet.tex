\cheatsheettitle{Algebra I Systems Cheatsheet}{Module 6}

\section*{Meaning of a Solution}
A solution to a system is an ordered pair that makes every equation true.
\[
\begin{cases}
y=2x+1\\
y=-x+7
\end{cases}
\]
The graphing solution is the intersection point.

\section*{Graphing Method}
\begin{itemize}
  \item Put both equations in graphable form.
  \item Graph both lines.
  \item The intersection is the solution.
\end{itemize}

\section*{Substitution Method}
Use when one equation is already solved for a variable.
\[
y=3x-4,\qquad 2x+y=11
\]
Substitute:
\[
2x+(3x-4)=11
\]

\section*{Elimination Method}
Use when adding equations can eliminate a variable.
\[
\begin{aligned}
2x+3y&=11\\
4x-3y&=7
\end{aligned}
\]
Add:
\[
6x=18
\]

\section*{Number of Solutions}
\[
\begin{array}{lll}
\text{one solution} & \text{different slopes} & \text{intersecting lines}\\
\text{no solution} & \text{same slope, different intercepts} & \text{parallel lines}\\
\text{infinitely many} & \text{same line} & \text{identical equations}
\end{array}
\]

\section*{Systems Word Problems}
Define variables first.
\[
x=\text{first quantity},\qquad y=\text{second quantity}
\]
Look for:
\[
\text{total equation},\qquad \text{rate/value equation}
\]

\section*{Quick Checks}
\begin{itemize}
  \item The answer to a two-variable system is usually an ordered pair.
  \item Substitute the ordered pair into both equations to check.
  \item Elimination works best when coefficients are already opposites or can be made opposites.
  \item Parallel lines have no intersection.
\end{itemize}
