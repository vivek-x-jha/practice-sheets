\section*{Formulas and Theorems}
\begin{itemize}[leftmargin=1.5em]
  \item Quadratic formula: $x = \dfrac{-b \pm \sqrt{b^{2}-4ac}}{2a}$; discriminant $b^2-4ac$ sets root type.
  \item Vertex of $y=ax^{2}+bx+c$: $x = -\dfrac{b}{2a}$, $y = f\!\left(-\dfrac{b}{2a}\right)$; $a<0$ gives a maximum, $a>0$ gives a minimum.
  \item Rational Root Theorem: any rational root of $a_nx^{n}+\dots+a_0$ is $\pm \dfrac{\text{factor of }a_0}{\text{factor of }a_n}$.
  \item Conjugate Root Theorem: non-real complex roots of polynomials with real coefficients occur in conjugate pairs $a\pm bi$.
  \item Remainder/Factor connection: dividing $f(x)$ by $(x-c)$ leaves remainder $f(c)$; if $f(c)=0$, $(x-c)$ is a factor.
  \item Polynomial long/synthetic division: $f(x) = (x-c)q(x) + r$ where $r=f(c)$.
  \item Factoring patterns: $A^{2}-B^{2}=(A-B)(A+B)$; $A^{3}\pm B^{3}=(A\pm B)(A^{2}\mp AB+B^{2})$.
  \item Exponent rules (for $a\neq0$): $a^{m}a^{n}=a^{m+n}$, $\dfrac{a^{m}}{a^{n}}=a^{m-n}$, $(a^{m})^{n}=a^{mn}$, $a^{-m}=\dfrac{1}{a^{m}}$.
  \item Absolute values with even roots: $\sqrt{x^{2}}=|x|$; include $|\,\cdot\,|$ unless the variable is restricted positive.
  \item Revenue/optimization: for quadratic $R(k)=ak^{2}+bk+c$ with $a<0$, maximum at $k=-\dfrac{b}{2a}$.
\end{itemize}
