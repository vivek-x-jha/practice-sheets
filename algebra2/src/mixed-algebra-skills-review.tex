\section*{Instructions}
Complete in 60 minutes. Show all work. Use exact forms unless a decimal is explicitly requested. For every ``simplify'' problem, write answers in fully simplified form: use only positive exponents, combine into a single fraction when possible, remove complex fractions, factor/cancel completely, and do not leave radicals in denominators.

\section*{Problems}
\begin{enumerate}
  \item Simplify:
  \[
  \frac{4x}{x^2-16}-\frac{3}{x+4}+\frac{5}{x-4}.
  \]

  \item Simplify:
  \[
  \frac{3}{2a^2b}-\frac{5}{6ab^2}+\frac{7}{3a^2b^2}.
  \]

  \item Simplify:
  \[
  \left(\frac{6x^2y^{-1}}{9z^3}\right)\left(\frac{12y^2z}{5x^{-1}}\right).
  \]

  \item Simplify:
  \[
  \frac{\frac{x^2-9}{x^2-2x-3}}{\frac{x^2-6x+9}{x^2-1}}.
  \]

  \item Combine like terms:
  \[
  7x^4-3x^2+9x-5+2x^4+11x^2-4x+8.
  \]

  \item Simplify:
  \[
  (2x^3y^{-2})^2(x^{-1}y^5)^3.
  \]

  \item Simplify:
  \[
  3\sqrt{75x^5y^3}-2\sqrt{12x^3y^5}+\sqrt{27x^5y^3}.
  \]

  \item Expand and simplify:
  \[
  2(3x-4)-5(2x+1)+3(x-7)-4(1-2x).
  \]

  \item Factor completely:
  \[
  6x^3y^2-15x^2y^4+9xy^3.
  \]

  \item Find the least common multiple of:
  \[
  18x^3y^2,\quad 30x^2y^5,\quad 42xy^4.
  \]

  \item Simplify to an expression with positive exponents only:
  \[
  \frac{(27a^{-3}b^6)^{2/3}}{(3a^{-1}b)^2}.
  \]

  \item Simplify:
  \[
  \frac{x^2-5x+6}{x^2-9}\cdot\frac{x^2+6x+9}{x^2-4x+3}\div\frac{x^2-1}{x^2-2x+1}.
  \]

  \item Find both the GCF and the LCM of:
  \[
  24x^4y^3z^2,\quad 36x^3y^5z,\quad 54x^2y^4z^3.
  \]

  \item Simplify:
  \[
  \frac{2}{x}-\frac{3}{x+h}
  \]
  and then simplify your result over $h$:
  \[
  \frac{\left(\frac{2}{x}-\frac{3}{x+h}\right)-\left(-\frac{1}{x}\right)}{h}.
  \]

  \item Simplify completely:
  \[
  \left(\frac{a^{-2}b^{3/2}}{c^{-1}d^{1/2}}\right)^{-2}
  \left(\frac{a^3c^2}{b^{-1}d}\right)
  \div
  \left(\frac{ab^{1/2}}{cd^{-1/2}}\right)^3.
  \]
\end{enumerate}
