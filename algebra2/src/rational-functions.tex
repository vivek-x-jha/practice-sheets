\section*{Instructions}
Complete in 60 minutes. Show clear work. Give exact forms and simplified fractions. State domain restrictions explicitly.

\section*{Problems}
\begin{enumerate}
  \item Simplify $f(x)=\dfrac{x^3-8}{x^2-4x}$ and list all domain restrictions and holes.
  \item Find all vertical asymptotes, horizontal/slant asymptotes, and holes of $g(x)=\dfrac{2x^3-3x^2-8x+12}{x^2-4x}$.
  \item Determine all $x$-intercepts, the $y$-intercept, and the end behavior of $h(x)=\dfrac{x^2-7x+10}{x^2+3x-10}$.
  \item Find the slant asymptote of $p(x)=\dfrac{4x^3-5x^2+2x-7}{x^2-3}$ and state all vertical asymptotes.
  \item Solve and check for extraneous solutions: $\dfrac{2x+1}{x-3}-\dfrac{x-4}{x+2}=3$.
  \item Solve and check for extraneous solutions: $\dfrac{3}{x-2}+\dfrac{4}{x+1}=\dfrac{5x-7}{x^2-x-2}$.
  \item Solve the inequality: $\dfrac{(x-1)(x+4)}{(x-3)(x+2)}\ge 0$.
  \item Write $\dfrac{5x^2+7x-3}{(x-1)(x+2)(x-3)}$ as a sum of partial fractions.
  \item For $r(x)=\dfrac{x^2+ax+b}{x^2-5x+6}$, find all $(a,b)$ so that $r(x)$ has a hole at $x=2$ and a vertical asymptote at $x=3$.
  \item For $s(x)=\dfrac{x^2-4x+k}{x^2-9}$, find all $k$ so that the graph has a horizontal asymptote at $y=1$ and exactly one removable discontinuity.
  \item Determine the domain and range of $t(x)=\dfrac{6}{x-4}-2$.
  \item A company’s average cost is modeled by $C(x)=\dfrac{x^2-10x+25}{x-5}+12$, where $x$ is the number of items (in hundreds). Find the domain, identify any holes or asymptotes, and simplify $C(x)$ for $x\ne 5$.
\end{enumerate}
