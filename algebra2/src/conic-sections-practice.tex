\section*{Instructions}
Complete in 60 minutes. Show clear work. Use exact forms unless a decimal approximation is explicitly requested.

\section*{Key Formulas}
\begin{itemize}
  \item Circle: $(x-h)^2+(y-k)^2=r^2$
  \item Parabola (vertical): $(x-h)^2=4p(y-k)$
  \item Parabola (horizontal): $(y-k)^2=4p(x-h)$
  \item Ellipse (major axis horizontal): $\dfrac{(x-h)^2}{a^2}+\dfrac{(y-k)^2}{b^2}=1,\; c^2=a^2-b^2,\; e=\dfrac{c}{a}$
  \item Ellipse (major axis vertical): $\dfrac{(x-h)^2}{b^2}+\dfrac{(y-k)^2}{a^2}=1,\; c^2=a^2-b^2,\; e=\dfrac{c}{a}$
  \item Hyperbola (transverse axis horizontal): $\dfrac{(x-h)^2}{a^2}-\dfrac{(y-k)^2}{b^2}=1,\; c^2=a^2+b^2,\; e=\dfrac{c}{a}$
  \item Hyperbola (transverse axis vertical): $\dfrac{(y-k)^2}{a^2}-\dfrac{(x-h)^2}{b^2}=1,\; c^2=a^2+b^2,\; e=\dfrac{c}{a}$
  \item Hyperbola asymptotes:
  \[
  y-k=\pm \frac{b}{a}(x-h)\quad \text{or} \quad y-k=\pm \frac{a}{b}(x-h)
  \]
\end{itemize}

\section*{Problems}
\begin{enumerate}
  \item Identify the conic and rewrite in standard form:
  \[
  9x^2+9y^2-54x+36y+36=0.
  \]

  \item Identify the conic and rewrite in standard form:
  \[
  4x^2-16x+y^2+6y-11=0.
  \]

  \item Identify the conic and rewrite in standard form:
  \[
  16x^2-9y^2+64x+36y-116=0.
  \]

  \item For
  \[
  (x+1)^2=12(y-4),
  \]
  find the vertex, focus, directrix, and axis of symmetry.

  \item For
  \[
  \frac{(x-3)^2}{49}+\frac{(y+2)^2}{9}=1,
  \]
  find the center, vertices, co-vertices, foci, and eccentricity.

  \item For
  \[
  \frac{(y-1)^2}{25}-\frac{(x+4)^2}{16}=1,
  \]
  find the center, vertices, foci, asymptotes, and eccentricity.

  \item Write an equation of the ellipse centered at the origin with vertices $(\pm 10,0)$ and foci $(\pm 8,0)$.

  \item Write an equation of the hyperbola centered at $(2,-1)$ with vertices $(2,4)$ and $(2,-6)$ and asymptote slope magnitude $\frac{5}{3}$.

  \item Write an equation of the parabola with focus $(5,-2)$ and directrix $x=-3$.

  \item Find the value(s) of $k$ such that
  \[
  4x^2+ky^2-16x+10y+1=0
  \]
  represents
  \begin{enumerate}
    \item an ellipse,
    \item a hyperbola,
    \item a parabola.
  \end{enumerate}

  \item Find all points of intersection of
  \[
  \frac{x^2}{16}+\frac{y^2}{4}=1
  \quad \text{and} \quad
  y=\frac{x}{2}.
  \]

  \item A satellite dish cross-section is modeled by
  \[
  y=\frac{1}{18}x^2
  \]
  (units in meters, vertex at the origin). Find the location of the focus and the depth-to-diameter ratio if the dish has diameter $9$ m.
\end{enumerate}
