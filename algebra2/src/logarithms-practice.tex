\section*{Formula Sheet}
\begin{itemize}[leftmargin=1.5em]
  \item Product: $\log_b(MN)=\log_b M+\log_b N$
  \item Quotient: $\log_b\!\left(\frac{M}{N}\right)=\log_b M-\log_b N$
  \item Power: $\log_b(M^k)=k\log_b M$
  \item Inverse: $\log_b(b^x)=x$ and $b^{\log_b x}=x$
  \item Change of base: $\log_b x=\dfrac{\log_a x}{\log_a b}$
  \item Exponential rules: $b^m b^n=b^{m+n}$, $\dfrac{b^m}{b^n}=b^{m-n}$, $(b^m)^n=b^{mn}$
  \item Common bases: $b=2,\ 10,\ e$ and $\ln x=\log_e x$
\end{itemize}

\section*{Problems}
\begin{enumerate}[leftmargin=*, itemsep=1.0em, topsep=0.8em]
  \item Find the root of the equation: $2+\lg\sqrt{1+x}+3\lg\sqrt{1-x}=\lg\sqrt{1-x^2}$.
  \item Determine the value of $x$ for $x>0$: $5^{2+4+6+\cdots+2x}=(0.04)^{-28}$.
  \item Find $x$ if:
  \[
  \begin{cases}
  4^{\frac{x}{y}+\frac{y}{x}}=32\\
  \log_3(x-y)+\log_3(x+y)=1
  \end{cases}
  \]
  \item Find $n$: $\log_{2}3\cdot\log_{3}4\cdot\log_{4}5\cdots\log_{n}(n+1)=10$.

  \item Expand: $\log_b\!\left(\dfrac{8x^4\sqrt[3]{y^2}}{b^3\sqrt{z}}\right)$.
  \item Condense: $\tfrac{3}{2}\log_b(x)-\log_b(y)+\tfrac{1}{4}\log_b(16)-2\log_b(b)$.
  \item Solve: $\log_2(x-2)+\log_2(x-6)=5$.
  \item Solve: $\log_5(2x+1)-\log_5(x-4)=2$.

  \item A culture grows according to $P(t)=P_0\cdot 3^{t}$, where $t$ is in hours. How many hours does it take for the population to multiply by 50?
  \item A sound has intensity $I=7.5\times10^{-6}\text{ W/m}^2$. Using $L=10\log_{10}\!\left(\frac{I}{10^{-12}}\right)$, find the sound level $L$ in decibels.
  \item The half-life of a substance is 5 years. How long will it take for the sample to decay to 12\% of its original amount?
  \item A city’s population is modeled by $P(t)=120000\cdot(1.08)^t$, where $t$ is in years. In what year will the population reach 300,000?
\end{enumerate}
