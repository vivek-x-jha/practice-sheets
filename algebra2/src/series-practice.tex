\section*{Instructions}
Complete in 60 minutes. Show clear work. Use exact values (fractions/radicals) unless a decimal is explicitly requested.

\section*{Problems}
\begin{enumerate}
  \item For the arithmetic sequence with $a_4=17$ and $a_{15}=61$, find $a_1$, $d$, and $a_{40}$.
  \item Find the exact value of $\sum_{k=1}^{35}\left(7-3k\right)$.
  \item The 12th term of an arithmetic sequence is 38 and the sum of the first 12 terms is 84. Find the first term and common difference.
  \item For the geometric sequence with $g_3=54$ and $g_7=\frac{2}{3}$, find $g_1$ and $r$ (both possible values of $r$).
  \item Evaluate exactly: $\sum_{n=0}^{8} 5\left(\frac{2}{3}\right)^n$.
  \item Find $n$ if $\sum_{k=1}^{n} 3\cdot 2^{k-1}=1533$.
  \item Write the sum as a telescoping expression and evaluate:
  \[
  \sum_{k=1}^{20}\frac{1}{k(k+1)}.
  \]
  \item Evaluate:
  \[
  \sum_{k=2}^{30}\left(\frac{1}{k-1}-\frac{2}{k}+\frac{1}{k+1}\right).
  \]
  \item Determine whether the infinite series converges or diverges, and if it converges, find the sum:
  \[
  \sum_{n=1}^{\infty}\left(\frac{4}{7}\right)^{n-1}.
  \]
  \item Determine whether the infinite series converges or diverges, and if it converges, find the sum:
  \[
  \sum_{n=0}^{\infty}(-3)\left(-\frac{1}{5}\right)^n.
  \]
  \item Find the exact sum:
  \[
  \sum_{k=1}^{\infty}\left(\frac{3}{2^{k}}-\frac{1}{3^{k}}\right).
  \]
  \item A ball is dropped from 54 meters and rebounds to $\frac{2}{3}$ of its previous height each time. Find the exact total vertical distance traveled after infinitely many bounces.
\end{enumerate}
