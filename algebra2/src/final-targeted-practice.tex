\section*{Instructions}
Complete in 60 minutes. Show clear work. Use exact values unless a decimal approximation is explicitly requested. For triangle measurements, round side lengths to the nearest tenth and angles to the nearest degree unless exact values are possible.

\section*{Reference Facts}
\begin{itemize}[leftmargin=1.5em]
  \item Principal inverse trig ranges:
  \[
  \sin^{-1}x\in\left[-\frac{\pi}{2},\frac{\pi}{2}\right],\qquad
  \cos^{-1}x\in[0,\pi],\qquad
  \tan^{-1}x\in\left(-\frac{\pi}{2},\frac{\pi}{2}\right)
  \]
  \item Law of Sines and Law of Cosines:
  \[
  \frac{a}{\sin A}=\frac{b}{\sin B}=\frac{c}{\sin C},\qquad
  c^2=a^2+b^2-2ab\cos C
  \]
  \item Triangle area:
  \[
  K=\frac12 ab\sin C
  \]
  \item Conic forms:
  \[
  (x-h)^2+(y-k)^2=r^2,\qquad (x-h)^2=4p(y-k),\qquad (y-k)^2=4p(x-h)
  \]
  \[
  \frac{(x-h)^2}{a^2}+\frac{(y-k)^2}{b^2}=1,\qquad
  \frac{(x-h)^2}{a^2}-\frac{(y-k)^2}{b^2}=1,\qquad
  \frac{(y-k)^2}{a^2}-\frac{(x-h)^2}{b^2}=1
  \]
\end{itemize}

\section*{Problems}
\begin{enumerate}[leftmargin=*, itemsep=1.15em, topsep=0.8em]
  \item Find the exact value of each expression.
  \[
  \sec\!\left(\tan^{-1}\!\left(-\frac{5}{12}\right)\right),\qquad
  \sin\!\left(\cos^{-1}\!\left(-\frac{7}{25}\right)\right),\qquad
  \cot\!\left(\sin^{-1}\!\left(\frac{4}{5}\right)\right)
  \]

  \item Solve on $0\le \theta<2\pi$.
  \[
  2\cos^2\theta-\cos\theta-1=0
  \]

  \item Solve on $0\le \theta<2\pi$.
  \[
  \sqrt{3}\sec\theta+2=0
  \]

  \item For
  \[
  y=-3\cos\left(2x+\frac{\pi}{2}\right)-1,
  \]
  state the amplitude, period, phase shift, midline, and range. Then give one interval that contains exactly one full period.

  \item Write an equation of a sine function with amplitude $2$, midline $y=-3$, period $4\pi$, and passing through the point $\left(\frac{\pi}{2},-3\right)$ at the beginning of a rising cycle.

  \item In $\triangle ABC$, $a=16$, $b=21$, and $C=58^\circ$. Find $c$, the area of the triangle, and the larger of $\angle A$ and $\angle B$.

  \item In $\triangle PQR$, $p=18$, $q=24$, and $P=42^\circ$. Determine how many triangles are possible. Solve every possible triangle.

  \item A hiker walks $5.2$ miles on a bearing of $038^\circ$, then $7.1$ miles on a bearing of $128^\circ$. How far is the hiker from the starting point?

  \item Identify the conic and rewrite in standard form.
  \[
  x^2-10x+8y-7=0
  \]
  Then state its vertex, focus, directrix, and axis of symmetry.

  \item Identify the conic and rewrite in standard form.
  \[
  9x^2+4y^2-54x+32y+109=0
  \]
  Then state the center and the lengths of the major and minor axes.

  \item Identify the conic and rewrite in standard form.
  \[
  25x^2-16y^2-200x-32y-16=0
  \]
  Then state the center, vertices, foci, and asymptotes.

  \item Write an equation of the hyperbola with center $(-2,3)$, vertices $(-2,8)$ and $(-2,-2)$, and asymptotes
  \[
  y-3=\pm \frac{5}{4}(x+2).
  \]
  Then find the coordinates of the foci.
\end{enumerate}
