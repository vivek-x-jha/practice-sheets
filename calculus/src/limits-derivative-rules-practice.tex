\section*{Basic Formulae}
\begin{itemize}[leftmargin=1.5em]
  \item Definition of derivative: $f'(x)=\displaystyle \lim_{h\to 0}\frac{f(x+h)-f(x)}{h}$
  \item Power rule: $\dfrac{d}{dx}(x^n)=nx^{n-1}$
  \item Product rule: $(fg)'=f'g+fg'$
  \item Quotient rule: $\left(\dfrac{f}{g}\right)'=\dfrac{f'g-fg'}{g^2}$
  \item Chain rule: $\dfrac{d}{dx}[f(g(x))]=f'(g(x))g'(x)$
  \item Trig derivatives:
  \[
  (\sin x)'=\cos x,\quad (\cos x)'=-\sin x,\quad (\tan x)'=\sec^2 x,
  \]
  \[
  (\cot x)'=-\csc^2 x,\quad (\sec x)'=\sec x\tan x,\quad (\csc x)'=-\csc x\cot x
  \]
  \item Special limits:
  \[
  \lim_{x\to 0}\frac{\sin x}{x}=1,\qquad
  \lim_{x\to 0}\frac{1-\cos x}{x}=0,\qquad
  \lim_{x\to 0}\frac{1-\cos x}{x^2}=\frac12
  \]
\end{itemize}

\section*{Problems}
\begin{enumerate}[leftmargin=*, itemsep=1em, topsep=0.8em]
  \item Evaluate:
  \[
  \lim_{x\to 0}\frac{\sin(7x)-7x}{x^3}.
  \]

  \item Evaluate:
  \[
  \lim_{x\to 0}\frac{1-\cos(5x)}{1-\cos(3x)}.
  \]

  \item Evaluate:
  \[
  \lim_{x\to 0}\frac{\sin(4x)\tan(3x)}{x^2}.
  \]

  \item Use the definition of the derivative to find $f'(x)$ for $f(x)=\dfrac{2}{x-1}$.

  \item Use the definition of the derivative to find $g'(x)$ for $g(x)=x^3-5x^2+2x-7$.

  \item Differentiate:
  \[
  y=(x^4-3x^2+1)(\sin x+\cos x+\tan x).
  \]

  \item Differentiate:
  \[
  y=\frac{(2x^2+1)(\sec x+\csc x)}{x^3-4x+2}.
  \]

  \item Differentiate:
  \[
  y=\sin\!\left((x^2-3x+1)^5\right).
  \]

  \item Differentiate:
  \[
  y=\sec(3x^2-x)\,\csc(x^3+1).
  \]

  \item Differentiate:
  \[
  y=(x^2+1)^4\cos\!\left(\frac{x^2-1}{x}\right).
  \]

  \item Differentiate:
  \[
  y=\left(\frac{x^2+3x-1}{x-2}\right)^6+\left(\frac{x-1}{x^2+1}\right)^5.
  \]

  \item Let
  \[
  f(x)=x^2\sin x,\qquad g(x)=\frac{\tan x}{x^2+1},\qquad h(x)=f(g(x)).
  \]
  Find $f'(x)$, $g'(x)$, and $h'(x)$.

  \item Evaluate:
  \[
  \lim_{x\to 0}\frac{\sin(2x)-\tan(2x)}{x^3}.
  \]

  \item Evaluate:
  \[
  \lim_{x\to 0}\frac{1-\cos(4x)}{x\sin(3x)}.
  \]

  \item Use the definition of the derivative to find $p'(x)$ for
  \[
  p(x)=\frac{1}{x^2+1}.
  \]

  \item Differentiate:
  \[
  y=(x^3+1)^7(\sin x)^3.
  \]

  \item Differentiate:
  \[
  y=\frac{(x^2-1)^4\cot x}{(x^3+2)^2}.
  \]

  \item Differentiate:
  \[
  y=\cos\!\left((3x^2-2x+5)^6\right)+\tan\!\left(\frac{1}{x^2+1}\right).
  \]

  \item Differentiate:
  \[
  y=\left[\frac{\sec x-\tan x}{x^2+4x+5}\right]^5.
  \]

  \item Let
  \[
  F(x)=x^3\cos x,\qquad G(x)=\frac{\csc x}{x^2-1},\qquad H(x)=F(G(x)).
  \]
  Find $F'(x)$, $G'(x)$, and $H'(x)$.
\end{enumerate}
