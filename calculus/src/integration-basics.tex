\section*{Instructions}
Evaluate or simplify each integral. Show key antiderivative steps. Use exact values; include +C for indefinite integrals.

\section*{Problems}
\begin{enumerate}
  \item $\displaystyle \int 5x^4 \, dx$
  \item $\displaystyle \int (3x^2 - 7x + 4) \, dx$
  \item $\displaystyle \int \left(6x^{\tfrac{1}{2}} - \dfrac{4}{x}\right) \, dx$
  \item $\displaystyle \int (2e^{x} - 5) \, dx$
  \item $\displaystyle \int \left(\dfrac{3}{\sqrt{x}} + 4x^3\right) \, dx$
  \item $\displaystyle \int \cos(2x) \, dx$
  \item $\displaystyle \int \left(7e^{-3x}\right) \, dx$
  \item $\displaystyle \int \left(4x^3 + \dfrac{2}{x^2}\right) \, dx$
  \item $\displaystyle \int \left(5\sin x + 3\cos x\right) \, dx$
  \item $\displaystyle \int_{0}^{2} (3x^2 - 2x + 1) \, dx$
  \item $\displaystyle \int_{1}^{e} \left(\dfrac{1}{x} + 2\right) \, dx$
  \item $\displaystyle \int_{-1}^{1} (x^3 - 4x) \, dx$
\end{enumerate}
