\section*{Instructions}
Show key derivative steps and supporting work. Use exact values unless a decimal approximation is explicitly requested.

\section*{Basic Formulae}
\begin{itemize}[leftmargin=1.5em]
  \item Scalar motion:
  \[
  v(t)=s'(t),\qquad a(t)=v'(t)=s''(t)
  \]
  \item Speed:
  \[
  \text{speed}=|v(t)|
  \]
  \item Vector motion:
  \[
  \mathbf{v}(t)=\mathbf{r}'(t),\qquad \mathbf{a}(t)=\mathbf{r}''(t)
  \]
  \item Speed in vector form:
  \[
  |\mathbf{v}(t)|=\sqrt{(x'(t))^2+(y'(t))^2+(z'(t))^2}
  \]
  \item A particle moves forward when $v(t)>0$ and backward when $v(t)<0$.
  \item A particle speeds up when velocity and acceleration have the same sign, and slows down when they have opposite signs.
\end{itemize}

\section*{Problems}
\begin{enumerate}[leftmargin=*, itemsep=1em, topsep=0.8em]
  \item A particle moves on a line with position
  \[
  s(t)=t^4-8t^3+18t^2+5,\qquad t\ge 0.
  \]
  Find $v(t)$ and $a(t)$. Then determine all intervals on which the particle is moving right, moving left, speeding up, and slowing down.

  \item A particle moves on a line with
  \[
  s(t)=t^3-6t^2+9t-4,\qquad 0\le t\le 5.
  \]
  Find all times when the particle changes direction. Then find the total distance traveled on $[0,5]$.

  \item A particle has velocity
  \[
  v(t)=3t^2-12t+5
  \]
  and position $s(0)=7$. Find $s(t)$. Then find the position and speed at $t=3$.

  \item A particle moves along a line so that
  \[
  s(t)=2\cos t+\sin(2t),\qquad 0\le t\le 2\pi.
  \]
  Find $v(t)$ and $a(t)$. Determine all times in $[0,2\pi]$ when the particle is momentarily at rest.

  \item The acceleration of a particle is
  \[
  a(t)=12t-18,
  \]
  with $v(1)=4$ and $s(1)=-3$. Find $v(t)$ and $s(t)$.

  \item A particle moves on a line with
  \[
  s(t)=\frac{t}{t^2+1},\qquad t\ge 0.
  \]
  Find $v(t)$ and $a(t)$. Then determine all $t>0$ for which the particle is moving left.

  \item A particle in the plane has position vector
  \[
  \mathbf{r}(t)=\langle t^2-3t,\ 2t^3-9t\rangle.
  \]
  Find $\mathbf{v}(t)$ and $\mathbf{a}(t)$. Then find the speed at $t=2$.

  \item A particle in the plane moves according to
  \[
  \mathbf{r}(t)=\langle e^t,\ e^{-t}\rangle.
  \]
  Find $\mathbf{v}(t)$, $\mathbf{a}(t)$, and $|\mathbf{v}(t)|$. Determine all $t$ for which the speed is $5$.

  \item A particle in space has position
  \[
  \mathbf{r}(t)=\langle \cos t,\ \sin t,\ t^2\rangle.
  \]
  Find $\mathbf{v}(t)$ and $\mathbf{a}(t)$. Then find the speed and acceleration magnitude at $t=\pi$.

  \item A particle moves in space with
  \[
  \mathbf{r}(t)=\langle t,\ t^2,\ t^3\rangle.
  \]
  Find the velocity and acceleration vectors at $t=1$. Then find the equation of the tangent line to the path at $t=1$ in symmetric form.

  \item A particle has position vector
  \[
  \mathbf{r}(t)=\langle 3\cos(2t),\ 3\sin(2t),\ 4t\rangle.
  \]
  Find $\mathbf{v}(t)$ and $\mathbf{a}(t)$. Then determine the angle between $\mathbf{v}\!\left(\frac{\pi}{4}\right)$ and $\mathbf{a}\!\left(\frac{\pi}{4}\right)$.

  \item A particle in the plane moves with
  \[
  \mathbf{r}(t)=\langle t^2+1,\ t^3-3t\rangle.
  \]
  Find the times when the particle is moving upward. Then find all times when the tangent to the path is horizontal.
\end{enumerate}
