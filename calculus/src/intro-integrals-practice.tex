\section*{Instructions}
Complete within 60 minutes. Use exact values; do not round. Show setup for Riemann sums, area, and particle motion problems.

\section*{Problems}
\begin{enumerate}[leftmargin=*, itemsep=1.0em, topsep=0.8em]
  \item The rate $R(t)$ (in thousands of people per day) of new sign-ups for a health program is recorded at the times below. Use a trapezoidal sum to approximate the total number of people after 36 days.
  \[
  \begin{array}{c|ccccccccc}
  t & 0 & 3 & 9 & 15 & 21 & 27 & 30 & 36\\ \hline
  R(t) & 1 & 2 & 4 & 7 & 6 & 5 & 8 & 9
  \end{array}
  \]

  \item Find the area between $y=1$ and $y=(x+1)^2-1$ on $[-2,1]$.

  \item Evaluate $\displaystyle \int_{-\pi/4}^{\pi/4} \frac{1}{2+\left(x\sqrt{2}\right)^2}\,dx$.

  \item Suppose $F(x)=\int_{1}^{x}(t^3-2t+5)^{1/2}\,dt$. Find $F'(x)$ and $F'(3)$.

  \item Write the Riemann sum
  \[
  \lim_{n\to\infty}\sum_{k=1}^n \left(2+\frac{3k}{n}+\left(\frac{3k}{n}\right)^2\right)\frac{2}{n}
  \]
  as a definite integral (do not evaluate).

  \item Use the midpoint rule with $n=6$ to approximate $\displaystyle \int_{0}^{6} (x^2+3x-1)\,dx$.

  \item Find $\dfrac{d}{dx}\left[\int_{-1}^{x^3}(5t^2-4t+1)\,dt\right]$.

  \item Use properties of derivatives to simplify $\dfrac{d}{dx}\big(5x^3-4x+\sqrt{x}\big)$ and $\dfrac{d}{dx}\big(\ln(x^2)\big)$ for $x>0$.

  \item A particle has velocity $v(t)=t^2-4t-1$ (m/s). Find the acceleration function, the time(s) when the particle changes direction, and the total distance traveled on $[0,5]$.

  \item A particle has velocity (m/s) given by:
  \[
  v(t)=
  \begin{cases}
  2t+1, & 0\le t\le 2\\
  5-t, & 2< t\le 5
  \end{cases}
  \]
  If $s(0)=0$, find $s(5)$ and the total distance traveled on $[0,5]$.

  \item A function $f$ is continuous and differentiable with $f'(x)=2f(x)-x$ and $f(0)=3$.
  (a) Find the tangent line to $y=f(x)$ at $x=0$.
  (b) Find $\dfrac{d}{dx}\big(3f(x)-x^2\big)$.

  \item Evaluate $\displaystyle \int_0^3 (6x-2)\,dx$ and interpret the result as signed area.
\end{enumerate}

\newpage
\section*{Solutions}
\begin{enumerate}[leftmargin=*, itemsep=0.8em, topsep=0.6em]
  \item Trapezoidal sum: $3\!\left(\frac{1+2}{2}\right)+6\!\left(\frac{2+4}{2}\right)+6\!\left(\frac{4+7}{2}\right)+6\!\left(\frac{7+6}{2}\right)+6\!\left(\frac{6+5}{2}\right)+3\!\left(\frac{5+8}{2}\right)+6\!\left(\frac{8+9}{2}\right)=198$ thousand.
  \item Area $=\displaystyle \int_{-2}^{-1+\sqrt{2}}\!\!\!\!\big(1-x^2-2x\big)\,dx+\int_{-1+\sqrt{2}}^{1}\!\!\!\!\big(x^2+2x-1\big)\,dx=\frac{1+8\sqrt{2}}{3}$.
  \item $\displaystyle \int_{-\pi/4}^{\pi/4}\frac{1}{2+(x\sqrt{2})^2}\,dx=\frac{1}{2}\arctan(x)\Big|_{-\pi/4}^{\pi/4}=\arctan\!\left(\frac{\pi}{4}\right)$.
  \item $F'(x)=\sqrt{x^3-2x+5}$, so $F'(3)=\sqrt{26}$.
  \item $\displaystyle \int_{0}^{2}\left(2+\frac{3}{2}x+\frac{9}{4}x^2\right)\,dx$.
  \item Midpoint rule ($n=6$, $\Delta x=1$): $M_6=\sum_{k=0}^{5} f\!\left(k+\tfrac{1}{2}\right)=119.5$.
  \item $\dfrac{d}{dx}\left[\int_{-1}^{x^3}(5t^2-4t+1)\,dt\right]=(5x^6-4x^3+1)\cdot 3x^2$.
  \item $\dfrac{d}{dx}(5x^3-4x+\sqrt{x})=15x^2-4+\dfrac{1}{2\sqrt{x}}$, and $\dfrac{d}{dx}\big(\ln(x^2)\big)=\dfrac{2}{x}$ for $x>0$.
  \item $a(t)=2t-4$. Direction changes at $t=2+\sqrt{5}$. Total distance on $[0,5]$ is $\frac{4+20\sqrt{5}}{3}$.
  \item $s(5)=\int_0^2(2t+1)\,dt+\int_2^5(5-t)\,dt=\frac{21}{2}$. Total distance: same since $v(t)\ge0$ on $[0,5]$.
  \item (a) $f'(0)=2f(0)-0=6$, tangent line: $y=6x+3$. (b) $\dfrac{d}{dx}(3f(x)-x^2)=3f'(x)-2x$.
  \item $\displaystyle \int_0^3(6x-2)\,dx=\left(3x^2-2x\right)\Big|_0^3=21$ (signed area $=21$).
\end{enumerate}
