\section*{Instructions}
Complete within 45 minutes. Show clear work. Use exact values in radical form when possible; otherwise round to the nearest hundredth.

\section*{Problems}
\begin{enumerate}[leftmargin=*, itemsep=1em]
  \item In right triangle $\triangle ABC$, $\angle B=90^\circ$, $\angle C=22^\circ$, and hypotenuse $AC=9.4$ m.
  \begin{enumerate}[label=\alph*)]
    \item Find $AB$ to the nearest hundredth.
    \item Find $BC$ to the nearest hundredth.
  \end{enumerate}

  \item A point is $240$ ft from the base of a flagpole. The angle of elevation to the top is $61^\circ$.
  \begin{enumerate}[label=\alph*)]
    \item Find the height of the pole to the nearest tenth.
    \item Find the line-of-sight distance from the point to the top to the nearest tenth.
  \end{enumerate}

  \item In right triangle $\triangle ABC$, $\angle B=90^\circ$, $AB=7\sqrt{3}$, and $\angle A=60^\circ$.
  \begin{enumerate}[label=\alph*)]
    \item Find $BC$ in simplest radical form.
    \item Find $AC$ in simplest radical form.
  \end{enumerate}

  \item In $\triangle ABC$, $\angle A=90^\circ$, $\angle C=45^\circ$, and $BC=18$.
  \begin{enumerate}[label=\alph*)]
    \item Find $AB$ in simplest radical form.
    \item Find $AC$ in simplest radical form.
  \end{enumerate}

  \item If $\sin(C)=\frac{5}{13}$ and $AC=26$ in a right triangle with right angle at $B$, find $\cos(C)$ as a fraction.

  \item The diagonal of a square is $30$ cm. Find the perimeter of the square in simplest radical form.

  \item The distance from each vertex of a regular pentagon to its center is $12$ in. Find the area of the pentagon to the nearest hundredth.

  \item An isosceles trapezoid has bases $12$ and $20$, and each base angle on the longer base is $58^\circ$. Find the perimeter to the nearest tenth.

  \item A triangle has side lengths $13$, $15$, and $20$. Find all three angle measures to the nearest tenth of a degree.

  \item In $\triangle ABC$, $AB=9$, $AC=6$, and $\angle C=36^\circ$. Determine how many triangles are possible, then solve for all possible values of $\angle A$.

  \item In $\triangle PQR$, $p=18$, $q=11$, and included angle $R=124^\circ$.
  \begin{enumerate}[label=\alph*)]
    \item Find side $r$ to the nearest hundredth.
    \item Find $\angle P$ and $\angle Q$ to the nearest tenth of a degree.
  \end{enumerate}

  \item From a point on level ground, the angle of elevation to the top of a building is $27^\circ$. After moving $45$ m closer, the angle becomes $41^\circ$. Find the building height to the nearest tenth.
\end{enumerate}
