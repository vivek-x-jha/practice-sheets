\section*{Instructions}
Complete within 55 minutes. Show all work. Use exact values when possible; otherwise round to the nearest tenth.

\section*{Problems}
\begin{enumerate}[leftmargin=*, itemsep=0.9em]
  \item Two hikers start at the same trailhead. One walks $8.4$ miles due east, the other walks $6.9$ miles at a bearing of $035^\circ$. Find the distance between the hikers.

  \item A surveyor measures two sides of a triangular lot as $210$ ft and $175$ ft, with included angle $58^\circ$. Find the third side and the area of the lot.

  \item A ship leaves port and travels $42$ km on a heading of $070^\circ$. It then travels $55$ km on a heading of $145^\circ$. How far is the ship from port?

  \item In $\triangle ABC$, $a=17$, $b=11$, and $\angle A=42^\circ$. Solve the triangle. State how many triangles are possible.

  \item In $\triangle PQR$, $p=15$, $q=23$, and $r=28$. Find $\angle P$, $\angle Q$, and $\angle R$.

  \item From point $A$, the angle of elevation to the top of a tower is $31^\circ$. From point $B$, which is 120 m closer to the tower along level ground, the angle of elevation is $54^\circ$. Find the tower's height.

  \item A drone is observed from two points on level ground that are 260 m apart. The angles of elevation are $39^\circ$ and $64^\circ$, and the drone is between the two observers. Find the drone's height.

  \item Prove:
  \[
  \frac{\sin x}{1+\cos x}+\frac{1+\cos x}{\sin x}=2\csc x.
  \]

  \item Prove:
  \[
  \frac{1-\cos x}{\sin x}=\frac{\sin x}{1+\cos x}.
  \]

  \item Prove:
  \[
  \sec^2 x-\tan^2 x=1.
  \]

  \item Solve on $0\le \theta<2\pi$:
  \[
  2\sin^2\theta-3\sin\theta+1=0.
  \]

  \item The first side of a triangle is 16 cm and the second side is 21 cm. The angle opposite the 16 cm side is $34^\circ$. Find all possible values of the angle opposite the 21 cm side and determine the number of possible triangles.
\end{enumerate}
